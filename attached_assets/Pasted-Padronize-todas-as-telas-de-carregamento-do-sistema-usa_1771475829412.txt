function mostrarLoadingApp(titulo, subtitulo, tempo = 4000) {

  document.getElementById("loadingTitle").innerText = titulo;
  document.getElementById("loadingSub").innerText = subtitulo;

  const box = document.getElementById("loadingFull");
  const bar = document.getElementById("progressFill");
  const txt = document.getElementById("progressText");

  box.style.display = "flex";

  document.body.style.overflow = "hidden";

  let progresso = 0;

  const intervalo = setInterval(() => {

    progresso += 1;

    bar.style.width = progresso + "%";
    txt.innerText = progresso + "%";

    if (progresso >= 100) {
      clearInterval(intervalo);
      esconderLoadingApp();
    }

  }, tempo / 100);
}

function esconderLoadingApp() {

  document.getElementById("loadingFull").style.display = "none";

  document.body.style.overflow = "auto";

  document.getElementById("progressFill").style.width = "0%";
  document.getElementById("progressText").innerText = "0%";
}
